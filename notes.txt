function force = forceCalculatation(clock)

if clock <= 10
    force = 1000;
elseif clock > 10 && clock <= 30
    force = -300;
elseif clock > 30 && clock <= 50
    force = 300;
else
    force = 0;
end

function steer = steerCalculation(clock)
